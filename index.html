                    analyzeIcon.classList.add('hidden');

                    loadingSpinner.classList.remove('hidden');

                    buttonText.textContent = 'Analyzing...';

                } else {

                    analyzeButton.disabled = false;

                    analyzeIcon.classList.remove('hidden');

                    loadingSpinner.classList.add('hidden');

                    buttonText.textContent = 'Analyze Idea';

                }

            };


            const displayError = (message) => {

                analysisOutput.classList.add('hidden');

                errorMessage.textContent = message || "An unknown error occurred. Please try again.";

                errorMessage.classList.remove('hidden');

            };


            const displayResults = (data) => {

                errorMessage.classList.add('hidden');

                

                // Clear previous results

                marketOutput.innerHTML = '';

                risksOutput.innerHTML = '';

                valuePropOutput.innerHTML = '';


                // Populate Target Market

                data.market.forEach(item => {

                    const li = document.createElement('li');

                    li.textContent = item;

                    marketOutput.appendChild(li);

                });


                // Populate Potential Risks

                data.risks.forEach(item => {

                    const li = document.createElement('li');

                    li.textContent = item;

                    risksOutput.appendChild(li);

                });


                // Populate Value Prop

                data.valueProp.forEach(item => {

                    const li = document.createElement('li');

                    li.textContent = item;

                    valuePropOutput.appendChild(li);

                });


                analysisOutput.classList.remove('hidden');

                // Scroll to results

                analysisOutput.scrollIntoView({ behavior: 'smooth', block: 'center' });

            };


            const callGeminiApi = async (userIdea, retries = 3, delay = 1000) => {

                setButtonLoading(true);

                analysisOutput.classList.add('hidden');

                errorMessage.classList.add('hidden');


                // --- Gemini API Configuration ---

                const apiKey = ""; // Will be injected by the environment

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;


                const systemPrompt = `You are an expert business consultant and strategist, trained by a 30-year veteran of Bloomberg and CNBC.

Analyze the following business idea. Provide a concise, professional, and insightful analysis.

For each of the three categories (market, risks, valueProp), provide 2-3 bullet points.

The output MUST be in the requested JSON format.`;


                const payload = {

                    contents: [{

                        parts: [{ text: userIdea }]

                    }],

                    systemInstruction: {

                        parts: [{ text: systemPrompt }]

                    },

                    generationConfig: {

                        responseMimeType: "application/json",

                        responseSchema: {

                            type: "OBJECT",

                            properties: {

                                "market": {

                                    "type": "ARRAY",

                                    "items": { "type": "STRING" }

                                },

                                "risks": {

                                    "type": "ARRAY",

                                    "items": { "type": "STRING" }

                                },

                                "valueProp": {

                                    "type": "ARRAY",

                                    "items": { "type": "STRING" }

                                }

                            },

                            required: ["market", "risks", "valueProp"]

                        }

                    }

                };

                // --- End of API Configuration ---


                try {

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: {

                            'Content-Type': 'application/json'

                        },

                        body: JSON.stringify(payload)

                    });


                    if (!response.ok) {

                        // Handle rate limiting (429) and other errors

                        if (response.status === 429 && retries > 0) {

                            await new Promise(res => setTimeout(res, delay));

                            return callGeminiApi(userIdea, retries - 1, delay * 2);

                        }

                        throw new Error(`API Error: ${response.status} ${response.statusText}`);

                    }


                    const result = await response.json();


                    if (result.candidates && result.candidates.length > 0 &&

                        result.candidates[0].content && result.candidates[0].content.parts &&

                        result.candidates[0].content.parts.length > 0) {

                        

                        const jsonText = result.candidates[0].content.parts[0].text;

                        const parsedData = JSON.parse(jsonText);

                        displayResults(parsedData);

                    } else {

                        throw new Error("Invalid response structure from API.");

                    }


                } catch (error) {

                    console.error("Gemini API call failed:", error);

                    displayError(error.message || "Failed to analyze idea. Please try again.");

                } finally {

                    setButtonLoading(false);

                }

            };

        });

    </script>
